#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for(int i = 0; i < n; i++) cin >> a[i];

        long long ans = 0;
        map<int, int> freq; // count of elements seen

        for(int i = n-1; i >= 0; i--) {
            // count elements <= a[i]
            auto it = freq.upper_bound(a[i]);
            long long count = 0;
            for(auto iter = freq.begin(); iter != it; iter++)
                count += iter->second;

            ans += count;

            // update frequency of a[i]
            freq[a[i]]++;
        }

        cout << ans << "\n";
    }

    return 0;
}
