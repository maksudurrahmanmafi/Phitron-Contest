#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    vector<string> programs(n);
    for (int i = 0; i < n; i++) cin >> programs[i];

    set<string> seen;
    vector<string> result;

    // reverse iterate
    for (int i = n - 1; i >= 0; i--) {
        if (seen.find(programs[i]) == seen.end()) {
            result.push_back(programs[i]);
            seen.insert(programs[i]);
        }
    }

    // print last 2 chars in order
    for (auto it = result.begin(); it != result.end(); ++it) {
        string s = *it;
        cout << s.substr(s.size() - 2, 2);
    }
    cout << "\n";

    return 0;
}
